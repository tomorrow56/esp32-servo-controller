# Dance Example
# 複数サーボで踊る

function pose_a {
  servos 0:45 1:135 2:45 3:135
  wait 300
}

function pose_b {
  servos 0:135 1:45 2:135 3:45
  wait 300
}

function pose_center {
  servos 0:90 1:90 2:90 3:90
  wait 300
}

# ダンス開始
call pose_center

repeat 8 {
  call pose_a
  call pose_b
}

call pose_center

# 回転動作
repeat 4 {
  servos 0:0 1:0 2:180 3:180
  wait 500
  servos 0:180 1:180 2:0 3:0
  wait 500
}

call pose_center
